#  列头符号转换。
> 例如： 1->A 26->Z   27->AA 28->AB
```js
function transfrom(number) {
    let res = ''
    number = number.toString(26)
    for (let i = 0; i < number.length; i++) {
        let char = number[i]
        if (i !== number.length - 1) {
            char--
        }
        char = parseInt(char, 26)
        res += String.fromCharCode(65 + char);
    }
    return res
}
```
